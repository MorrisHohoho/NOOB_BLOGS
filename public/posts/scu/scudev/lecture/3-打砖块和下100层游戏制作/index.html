#define GOLD RGB(255,215,0)
#define GARY RGB(200,200,200)
#define DARK RGB(0, 128, 128)
#define BLUE RGB(0, 0, 255)


#include <graphics.h>
#include <conio.h>
#include <stdio.h>
#include<iostream>
#include<random>
using namespace std;

default_random_engine e;
uniform_real_distribution<float> u(0.0, 10.0);
int item_x, item_y;		//é“å…·æ¨ªçºµåæ ‡
bool lenth_change = false;


//ç»˜åˆ¶é“å…·
bool add_item(int x, int y)
{
	item_x = x, item_y = y;
	int width = 40, height = 40;
	setlinecolor(GOLD);
	setfillcolor(GOLD);
	fillrectangle(item_x, item_y, item_x + width, item_y + height);
	return true;
}

//åŠ é•¿æŒ¡æ¿
void add_lenth(int bx, int& len)
{
	if (!lenth_change)
	{
		int item_vy = 10;	//é“å…·ä¸‹è½é€Ÿåº¦
		int width = 40, height = 40;
		item_y += item_vy;
		setlinecolor(GOLD);
		setfillcolor(GOLD);
		fillrectangle(item_x, item_y, item_x + width, item_y + height);
		if (item_y + height > 760 && item_x + width > bx && item_x < bx + len)
		{
			len += 50;
			lenth_change = true;
			setlinecolor(GARY);
			setfillcolor(GARY);
			fillrectangle(item_x, item_y, item_x + width, item_y + height);
		}
	}
}

int main()
{
	int r = 10;
	int x = 300, y = 750;	//å°çƒåæ ‡
	int vx = 8, vy = -10;	//å°çƒé€Ÿåº¦
	int i, j;
	int w = 90, h = 40;		//ç –å—é•¿ã€å®½
	int xs, ys;
	int gap = 5;	//ç –å—é—´è·
	int bx = 250, len = 100, th = 15;	//æ¿å­é•¿å®½
	bool s = false;		//æ¸¸æˆå¼€å§‹æ ‡å¿—
	bool f = false;		//æŒ‰ç©ºæ ¼é‡æ–°å¼€å§‹
	bool get_item = false;	//åˆ¤æ–­å°çƒæ˜¯å¦æ‹¾å–é“å…·item
	bool bricks_hit[128] = { false };	//åˆ¤æ–­å¯¹åº”ç –å—æ˜¯å¦è¢«æ’å‡»
	int count = 0;	//æ’å‡»ç –å—æ€»æ•°
	int brick_left, brick_top, brick_right, brick_bottom;

	initgraph(600, 800);

	setbkcolor(GARY);
	setlinecolor(DARK);
	setfillcolor(DARK);
	cleardevice();

	BeginBatchDraw();	//æ‰¹é‡ç»˜åˆ¶ï¼Œè®©ç”»é¢æ›´åŠ ä¸æ»‘
	while (1)
	{
	
		if (get_item)
		{
			add_lenth(bx, len);
		}


		setfillcolor(DARK);
		for (i = 0; i < 6; i++) // create a 6 * 5 matrix
			for (j = 0; j < 5; j++)
			{
				setfillcolor(DARK);
				setlinecolor(DARK);
			
				//èƒœåˆ©æç¤º
				if (count == 30)
				{
					settextcolor(RGB(0, 0, 255));
					settextstyle(60, 0, _T("Arial"));
					outtextxy(270, 400, _T("Win"));
					break;
				}

				//ç –å—åæ ‡è®¡ç®—
				xs = i * w;
				ys = j * h;
				brick_left = gap * (i + 1) + xs;
				brick_top = gap * (j + 1) + ys;
				brick_right = gap * (i + 1) + xs + w;
				brick_bottom = gap * (j + 1) + ys + h;

				//å°çƒæ’å‡»ç –å—ä¸Šä¸‹é¢
				if ((y - r < brick_bottom && x + r<brick_right && x - r > brick_left) && brick_bottom - y <= h)
				{
					if (!bricks_hit[i * 10 + j])
					{
						bricks_hit[i * 10 + j] = true;
						vy = -vy;
						count++;

						//50%çš„æ¦‚ç‡æ‰è½é“å…·ï¼Œåœ¨æ‰“æ‰ç¬¬15ä¸ªç –å—æ—¶ä¸€å®šæ‰è½ä¸€ä¸ª
						if (u(e) < 5 || count == 15)
						{
							get_item = add_item(brick_left, brick_top);
						}
					}
				}

				//å°çƒæ’å‡»ç –å—å·¦å³é¢
				if (x - r < brick_right && y + r<brick_bottom && y - r>brick_top && x - brick_left <= w && x - brick_left>0)
				{
					if (!bricks_hit[i * 10 + j])
					{
						bricks_hit[i * 10 + j] = true;
						vx = -vx;
						count++;

						if (u(e) < 5 || count == 15)
						{
							get_item = add_item(brick_left, brick_top);
						}
					}
				}

				//å¦‚æœç –å—è¢«æ’å‡»ï¼Œç”¨åº•è‰²è¦†ç›–
				if (bricks_hit[i * 10 + j])
				{
					setfillcolor(GARY);
					setlinecolor(GARY);
					fillrectangle(gap * (i + 1) + xs, gap * (j + 1) + ys, gap * (i + 1) + xs + w, gap * (j + 1) + ys + h);
					continue;
				}
				fillrectangle(gap * (i + 1) + xs, gap * (j + 1) + ys, gap * (i + 1) + xs + w, gap * (j + 1) + ys + h);
			}

		setfillcolor(RGB(0, 0, 128));
		fillrectangle(bx, 760, bx + len, 760 + th); // draw the board

		if (_kbhit())
		{
			char input = _getch();
			if (input == ' ')
			{
				s = true; // game start
				f = true;
			}
			if (GetAsyncKeyState(VK_LEFT))
			{
				bx = bx - 8;
				if (bx <= 0)
					bx = 0;
			}
			if (GetAsyncKeyState(VK_RIGHT))
			{
				bx = bx + 8;
				if (bx >= 600)
					bx = 600;
			}
		}
		if (!s)  //  before game has been started
		{
			setfillcolor(RGB(128, 0, 0));
			fillcircle(bx + len / 2, 760 - r, r);
		}
		else  //  after game has been started
		{
			if ((((y > 750) && (y < 800)) && (x + r >= bx) && (x - r <= bx + len)) || y < r)
				vy = -vy;
			if ((x > 600 - r) || (x < r))
				vx = -vx;
			if (f) // initial position of the ball
			{
				x = bx + len / 2;
				y = 760 - r;
				f = !f;
			}
			else  // the position of the ball during the game
			{
				x = x + vx;
				y = y + vy;
				if (y >= 800 && count < 30)
				{
					settextcolor(RGB(0, 0, 255));
					settextstyle(60, 0, _T("Arial"));
					outtextxy(270, 400, _T("Lose"));
					// break;
				}
			}
			setfillcolor(RGB(128, 0, 0));
			fillcircle(x, y, r);
		}

		FlushBatchDraw();
		Sleep(30);
		cleardevice();
	}
	closegraph();
	return 0;
}
